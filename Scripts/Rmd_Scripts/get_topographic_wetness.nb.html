function to add ngr



add.ngr<- function(data){
 points <- sgo_points(data, coords = c("x", "y"), epsg=27700)
points2 <- sgo_bng_ngr(points, digits = 4)
data <- data %>%
  mutate(ngr.1km = points2$ngr) %>%
  mutate(ngr.1km =  str_replace_all(ngr.1km, " ", ""))
  return(data)
}
